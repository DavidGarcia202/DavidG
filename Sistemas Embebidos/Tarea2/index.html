#define Led2 7   //

#define Led4 9   // (Most Significant Bit)

    // M√°scara con los 4 LEDs

            contador = 0;
        }
    }
}

    Secuencia en codigo Gray

    • Mostrar la secuencia del 0 al 15 en forma de c√≥digo Gray

    C√≥digo

    #include "pico/stdlib.h"

#define Led1 6   // (LSB)
#define Led2 7   
#define Led3 8   
#define Led4 9   // (MSB)

int main() {
    const uint32_t Mascara = (1u << Led1 | 1u << Led2 | 1u << Led3 | 1u << Led4);

    gpio_init_mask(Mascara);
    gpio_set_dir_masked(Mascara, Mascara);

    int contador = 0;

    while (true) {
        // Convertir de binario a Gray
        int gray = contador ^ (contador >> 1);

        // Mandar a los LEDs
        gpio_put_masked(Mascara, (gray << Led1));

        sleep_ms(1000);

        contador++;
        if (contador > 15) {
            contador = 0;
        }
    }
}
